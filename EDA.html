what items are over 5000 calories each?

cal_dense = 
  menustat_df %>% 
  mutate(
    as.numeric(calories)
  ) %>% 
  filter(
    calories > 5000
  )

fast_cal_dense = 
  cal_dense %>% 
  filter(
    rest_type == "fast food"
  ) %>% 
  select(rest_type, item_name, calories)

fast_cal_dense_plot =
  fast_cal_dense %>% 
  ggplot(aes(x = item_name, y = calories)) +
  geom_col() +
  theme(axis.text.x = element_text(angle = 65, hjust = 1, size = 6)) +
  labs(
    x = "Item Name",
    y = "Calories per Item",
    title = "Food Items > 5500 calories in Fast Food Restaurants"
  ) 

fast_cal_dense_plot

sit_cal_dense = 
  cal_dense %>% 
  filter(
    rest_type == "sit-down"
  ) %>% 
  select(rest_type, item_name, calories)
